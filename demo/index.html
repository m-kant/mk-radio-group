mk-radio-group

Basic usage

HTML
<input id='rGroup1' />
JavaScript
var options = {
	value: 5,
	enum: [
		{ value: 1, name: 'first item' },
		{ value: 2, name: 'second item' },
		{ value: 3, name: 'third item', },
		{ value: 4, name: 'fourth item of very important list' },
		{ value: 5, name: 'fifth item' },
		{ value: 6, name: 'sixth item',},
	]
}
var RG = new mkRadioGroup(rGroup1, options);

Data in attributes

HTML
<input id='rGroup2' value='5' data-enum='["1","2","3","4","5","6","7","8"]' />
JavaScript
var RG2 = new mkRadioGroup(rGroup2);

Stacked

Selected value is always scrolled into view

HTML
<input id='rGroup3' value='eight' data-enum='["one","two","three","four","five","six","seven","eight"]' />
JavaScript
var RG3 = new mkRadioGroup(rGroup3, {stacked: true});
